#       FLTKCONFIG = /usr/local/bin/fltk-config
#       FLTKCONFIG = /usr/local/src/fltk-1.3.x-svn/fltk-config

# Set .SILENT in your Makefile if you want 'quieter' builds. #

ifeq '$(OS)' "Windows_NT"
EXEEXT = .exe
endif

FLTKCONFIG = ../fltk-config 

CXX          = $(shell $(FLTKCONFIG) --cxx)
CXXFLAGS     = $(shell $(FLTKCONFIG) --cxxflags) -Wall -I.
LINKFLTK     = $(shell $(FLTKCONFIG) --ldstaticflags)

LINKFLTK_GL = $(shell $(FLTKCONFIG) --use-gl --ldstaticflags) LINKFLTK_IMG = $(shell $(FLTKCONFIG) --use-images --ldstaticflags) LINKFLTK_ALL = $(shell $(FLTKCONFIG) --use-images --use-gl --ldstaticflags) .SUFFIXES: .cxx .h .fl .o $(EXEEXT)

# HOW TO COMPILE
.cxx.o: 

        @echo "*** Compile $<..."
        $(CXX) -I.. $(CXXFLAGS) -c $< -o $@

# HOW TO LINK
.o$(EXEEXT): 

        @echo "*** Linking $@..."
        $(CXX) $< $(LINKFLTK_ALL) -o $@
